```js
// 全局通用脚本：页面加载完成后执行
document.addEventListener('DOMContentLoaded', function() {
    // 导航栏激活状态（可选）
    const currentPath = window.location.pathname;
    const navLinks = document.querySelectorAll('.nav-links a');
    navLinks.forEach(link => {
        if (link.href.includes(currentPath)) {
            link.style.color = '#ff6b6b';
            link.style.fontWeight = '700';
        }
    });

    // 页面加载动画（可选）
    document.body.style.opacity = '0';
    document.body.style.transition = 'opacity 0.5s ease';
    setTimeout(() => {
        document.body.style.opacity = '1';
    }, 100);
});
```

```js
// 旅游记录图片预览脚本
document.addEventListener('DOMContentLoaded', function() {
    const images = document.querySelectorAll('.travel-image');
    // 点击图片放大预览（简单实现）
    images.forEach(img => {
        img.addEventListener('click', function() {
            const preview = document.createElement('div');
            preview.style.position = 'fixed';
            preview.style.top = '0';
            preview.style.left = '0';
            preview.style.width = '100vw';
            preview.style.height = '100vh';
            preview.style.backgroundColor = 'rgba(0,0,0,0.8)';
            preview.style.display = 'flex';
            preview.style.justifyContent = 'center';
            preview.style.alignItems = 'center';
            preview.style.zIndex = '9999';
            preview.style.cursor = 'pointer';

            const previewImg = document.createElement('img');
            previewImg.src = img.src;
            previewImg.style.maxWidth = '90%';
            previewImg.style.maxHeight = '90%';

            preview.appendChild(previewImg);
            document.body.appendChild(preview);

            // 点击关闭预览
            preview.addEventListener('click', function() {
                document.body.removeChild(preview);
            });
        });
    });
});
```

```py
# 导入socket模块（原代码中使用了socket相关方法但未导入，补充缺失依赖）
import socket
# 导入基础HTTP请求处理器：处理普通HTTP请求（静态文件等）
from http.server import SimpleHTTPRequestHandler
# 导入CGI请求处理器：支持执行CGI脚本（动态请求）
from http.server import CGIHTTPRequestHandler
# 导入多线程HTTP服务端：支持并发处理请求，避免单线程阻塞
from http.server import ThreadingHTTPServer
# 导入偏函数：用于给处理器类绑定固定参数（如指定服务目录）
from functools import partial
# 导入上下文管理工具：用于优雅地捕获并忽略指定异常
import contextlib
# 导入系统模块：用于程序的正常退出
import sys
# 导入操作系统模块：用于获取当前工作目录、路径相关操作
import os


# 自定义双栈服务端类，继承自多线程HTTP服务端ThreadingHTTPServer
# 双栈指同时支持IPv4和IPv6协议，解决单协议绑定的限制
class DualStackServer(ThreadingHTTPServer):
    # 重写服务端绑定方法：核心是设置IPv6兼容IPv4
    def server_bind(self):
        # 捕获并忽略所有异常（主要是IPv4环境下设置IPv6参数的异常）
        # 目的是让服务端在纯IPv4环境下也能正常运行，不因为IPv6参数报错
        with contextlib.suppress(Exception):
            # 设置IPv6套接字选项：关闭IPv6独用模式（IPV6_V6ONLY=0）
            # 这样同一个端口可以同时监听IPv4和IPv6的请求
            self.socket.setsockopt(socket.IPPROTO_IPV6, socket.IPV6_V6ONLY, 0)
        # 调用父类的server_bind方法，执行默认的端口/地址绑定逻辑
        return super().server_bind()


def run(
    server_class=DualStackServer,    # 服务端类，默认使用自定义的双栈服务端
    handler_class=SimpleHTTPRequestHandler,  # 请求处理器类，默认处理普通HTTP请求
    port=8000,                       # 服务监听端口，默认8000
    bind='127.0.0.1',                # 服务绑定地址，默认本地回环（仅本机可访问）
    cgi=False,                       # 是否开启CGI模式，默认关闭（仅静态服务）
    directory=os.getcwd()            # 服务根目录，默认当前运行脚本的目录
):
    """
    启动HTTP服务端的核心函数
    Args:
        server_class: 服务端实现类，需继承自HTTPServer
        handler_class: 请求处理器类，需继承自BaseHTTPRequestHandler
        port: 监听端口，int类型，建议1024以上（1024以下需管理员权限）
        bind: 绑定地址，str类型；设为0.0.0.0则全网可访问，::则支持IPv6全网访问
        cgi: 是否开启CGI支持，开启后可执行指定目录下的CGI脚本处理动态请求
        directory: 服务的根目录，所有静态文件请求均基于此目录查找
    """
    # 判断是否开启CGI模式，选择对应的请求处理器并绑定服务目录
    if cgi:
        # CGI模式：使用CGI处理器，通过partial绑定directory参数
        # partial的作用：避免手动继承类传参，直接给处理器类固定目录属性
        handler_class = partial(CGIHTTPRequestHandler, directory=directory)
    else:
        # 普通模式：使用静态文件处理器，同样绑定directory参数
        handler_class = partial(SimpleHTTPRequestHandler, directory=directory)

    # 创建服务端实例并通过with上下文管理（自动处理资源释放）
    # 参数1：元组(bind, port) 绑定的地址和端口；参数2：处理请求的处理器类
    with server_class((bind, port), handler_class) as httpd:
        # 打印服务启动信息，告知用户访问地址
        print(
            f"Serving HTTP on {bind} port {port} "
            f"(http://{bind}:{port}/) ..."
        )
        try:
            # 启动服务并永久运行，直到手动终止（Ctrl+C）
            httpd.serve_forever()
        except KeyboardInterrupt:
            # 捕获键盘中断异常（Ctrl+C），优雅退出
            print("\nKeyboard interrupt received, exiting.")
            # 正常退出程序，退出码0表示无错误
            sys.exit(0)


# 程序入口：仅当直接运行该脚本时，才执行启动逻辑
if __name__ == '__main__':
    # 调用run函数启动服务，默认配置：本地8000端口、静态服务、当前目录
    run(port=8000, bind='127.0.0.1')
```
